// 지역 이동 시스템
public class RegionTransitionManager
{
    public void MoveToRegion(Region destination, Vehicle travelMethod)
    {
        // 이동 거리/시간 계산
        float travelTime = CalculateTravelTime(destination, travelMethod);

        // 로딩 시퀀스
        StartLoadingSequence(destination, travelTime);

        // 목적지 도착 이벤트
        OnArrivalEvent(destination);
    }

    // 경로별 시간 계산
    public Dictionary<(Region, Region), float> TravelTimes = new()
    {
        { (Erebos, Argentium), 48f }, // 약 2일
        { (Erebos, Sylvanheim), 72f }, // 약 3일
        { (Argentium, NorthernPermafrost), 72f }, // 약 3일
        // ... 기타 경로
    };
}
